class profesor

def intialize
	@min = 0
	@max = sillas-1
	@numalumnos = 0
	@lugares = Concurrent::Semaphore.new(4)
	@mux = Concurrent::Semaphore.new(0)
	@mux1 = Concurrent::Semaphore.new(0)
end

def pasar
	@mux1.acquire
	if (@numalumnos <= @max)
		@lugares.acquire
		@numalumnos = @numalumnos+1
	end
	@mux1.release
	return pregunta
end

def pregunta
	@mux.release
	puts "le estoy preguntando"
	sleep 0.3
	@mux.acquire
	@lugares.release
end

def descanso 
	if(@numalumnos == 0)
		puts "estoy descanzando"
		sleep 0.3
	end
	else
		return pasar
	end
end
